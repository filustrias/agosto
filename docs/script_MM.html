Aqui está o código para produção da tabela rj-sms-sandbox.sub_pav_us.MM_2026_novos_cadastros_stopp_start

```{sql connection=}

– ============================================================================ – TABELA: MM_EXAMES_E_RISCO – ============================================================================ – Objetivo: Consolidação por PACIENTE (1 linha/CPF) com: – - Dados demográficos e de cadastro – - 56 Morbidades (primeira data de diagnóstico) – - HIV (base externa + CD4/CV) – - Prescrições agregadas – - Exames laboratoriais (perfil lipídico, glicemia, creatinina) – - eGFR e estadiamento CKD – - Escore de Framingham – - Risco cardiovascular – - Dislipidemia (CID + laboratorial) – ============================================================================

CREATE OR REPLACE TABLE rj-sms-sandbox.sub_pav_us.MM_2026_novos_cadastros_stopp_start AS

WITH – ============================================================================ – SEÇÃO 1: IDENTIFICAÇÃO DE HIV/AIDS (Base Externa + CD4/CV) – ============================================================================

hiv_cpf_raw AS ( SELECT h.id_prontuario_global, h.id_cnes, h.aids_doenca_atual_ano_1_teste AS ano_diagnostico, a.patient_cpf AS patient_cpf_raw, REGEXP_REPLACE(CAST(a.patient_cpf AS STRING), r’‘,’‘) AS cpf FROM rj-sms.brutos_prontuario_vitacare_historico.hiv AS h JOIN rj-sms.brutos_prontuario_vitacare_historico.acto AS a USING (id_prontuario_global) WHERE NULLIF(TRIM(CAST(a.patient_cpf AS STRING)),’’) IS NOT NULL ),

hiv_cpf AS ( SELECT * FROM hiv_cpf_raw WHERE LENGTH(cpf) = 11 QUALIFY ROW_NUMBER() OVER ( PARTITION BY cpf ORDER BY id_prontuario_global DESC ) = 1 ),

exames_hiv_cpf_raw AS ( SELECT ea.id_hci, ea.entrada_data, ea.paciente_cpf AS paciente_cpf_raw, er.tipo, er.descricao, REGEXP_REPLACE(CAST(ea.paciente_cpf AS STRING), r’‘,’‘) AS cpf FROM rj-sms.saude_historico_clinico.episodio_assistencial ea CROSS JOIN UNNEST(ea.exames_realizados) er WHERE ea.exames_realizados IS NOT NULL AND ARRAY_LENGTH(ea.exames_realizados) > 0 AND er.descricao IS NOT NULL AND REGEXP_CONTAINS(er.descricao, r’(?i)cd4|cv|carga viral)) AND NULLIF(TRIM(CAST(ea.paciente_cpf AS STRING)), ’’) IS NOT NULL ),

exames_hiv_cpf_last AS ( SELECT * FROM exames_hiv_cpf_raw WHERE LENGTH(cpf) = 11 QUALIFY ROW_NUMBER() OVER ( PARTITION BY cpf ORDER BY entrada_data DESC, id_hci DESC ) = 1 ),

hiv_completo AS ( SELECT h.cpf, h.id_prontuario_global, h.id_cnes, h.ano_diagnostico, e.entrada_data AS data_ultimo_exame_hiv,

-- Data de diagnóstico (ano convertido para data) - COM VALIDAÇÃO
CASE 
  WHEN h.ano_diagnostico IS NOT NULL 
       AND SAFE_CAST(h.ano_diagnostico AS INT64) IS NOT NULL
       AND SAFE_CAST(h.ano_diagnostico AS INT64) >= 1980  -- HIV surgiu nos anos 80
       AND SAFE_CAST(h.ano_diagnostico AS INT64) <= EXTRACT(YEAR FROM CURRENT_DATE())
  THEN DATE(SAFE_CAST(h.ano_diagnostico AS INT64), 1, 1)
  ELSE NULL
END AS data_diagnostico_hiv

FROM hiv_cpf h LEFT JOIN exames_hiv_cpf_last e USING (cpf) ),

– ============================================================================ – SEÇÃO 2: CAPTURA DE EXAMES LABORATORIAIS (3 FONTES) – ============================================================================

exames_historico AS ( SELECT paciente_cpf, exame.codigo_do_exame AS cod_exame, exame.unidade, SAFE_CAST(exame.resultado AS FLOAT64) AS resultado_num, SAFE_CAST(PARSE_DATETIME(‘%d/%m/%Y %H:%M:%S’, exame.data_do_exame) AS DATE) AS data_exame FROM rj-sms.saude_historico_clinico.exame, UNNEST(exames) AS exame WHERE exame.resultado IS NOT NULL AND exame.codigo_do_exame IN (‘CRE’, ‘COL’, ‘HDL’, ‘LDL’, ‘TRI’, ‘GLI’, ‘GLU’, ‘HGL’) AND paciente_cpf IS NOT NULL AND SAFE_CAST(exame.resultado AS FLOAT64) > 0 AND SAFE_CAST(exame.resultado AS FLOAT64) < 10000 ),

exames_cientifica AS ( WITH

solicitacoes_paciente AS (
    SELECT 
        s.id AS solicitacao_id,
        REGEXP_REPLACE(CAST(s.paciente_cpf AS STRING), r'\D', '') AS cpf,
        s.unidade
    FROM `rj-sms.brutos_cientificalab.solicitacoes` s
    WHERE s.paciente_cpf IS NOT NULL
      AND LENGTH(REGEXP_REPLACE(CAST(s.paciente_cpf AS STRING), r'\D', '')) = 11
),

exames_processados AS (
    SELECT 
        e.id AS exame_id,
        e.solicitacao_id,
        e.cod_apoio,
        e.data_assinatura
    FROM `rj-sms.brutos_cientificalab.exames` e
    WHERE e.solicitacao_id IS NOT NULL
      AND e.data_assinatura IS NOT NULL
      AND e.cod_apoio IN ('CRE', 'COL', 'HDL', 'LDL', 'TRI', 'GLI', 'GLU', 'HGL')
),

resultados_exames AS (
    SELECT 
        r.exame_id,
        r.cod_apoio,
        r.resultado,
        SAFE_CAST(r.resultado AS FLOAT64) AS resultado_num,
        r.unidade
    FROM `rj-sms.brutos_cientificalab.resultados` r
    WHERE r.exame_id IS NOT NULL
      AND r.resultado IS NOT NULL
      AND r.cod_apoio IN ('CRE', 'COL', 'HDL', 'LDL', 'TRI', 'GLI', 'GLU', 'HGL')
      AND SAFE_CAST(r.resultado AS FLOAT64) > 0
      AND SAFE_CAST(r.resultado AS FLOAT64) < 10000
)

SELECT 
    sp.cpf AS paciente_cpf,
    COALESCE(re.cod_apoio, ep.cod_apoio) AS cod_exame,
    re.unidade,
    re.resultado_num,
    SAFE_CAST(ep.data_assinatura AS DATE) AS data_exame
FROM solicitacoes_paciente sp
INNER JOIN exames_processados ep 
    ON sp.solicitacao_id = ep.solicitacao_id
INNER JOIN resultados_exames re 
    ON ep.exame_id = re.exame_id

),

todos_exames AS ( SELECT * FROM exames_historico UNION ALL SELECT * FROM exames_cientifica –UNION ALL –SELECT * FROM exames_laboratorial ),

exames_filtrados AS ( SELECT paciente_cpf, cod_exame, resultado_num, data_exame FROM todos_exames WHERE (cod_exame = ‘COL’ AND resultado_num BETWEEN 50 AND 600) OR (cod_exame = ‘HDL’ AND resultado_num BETWEEN 10 AND 150) OR (cod_exame = ‘LDL’ AND resultado_num BETWEEN 20 AND 400) OR (cod_exame = ‘TRI’ AND resultado_num BETWEEN 20 AND 1000) OR (cod_exame = ‘GLI’ AND resultado_num BETWEEN 30 AND 500) OR (cod_exame = ‘GLU’ AND resultado_num BETWEEN 30 AND 500) OR (cod_exame = ‘HGL’ AND resultado_num BETWEEN 3 AND 20 AND unidade = ‘%’) OR (cod_exame = ‘CRE’ AND resultado_num BETWEEN 0.2 AND 15 AND unidade = ‘mg/dL’) ),

exames_mais_recentes AS ( SELECT paciente_cpf, cod_exame, resultado_num, data_exame FROM exames_filtrados QUALIFY ROW_NUMBER() OVER ( PARTITION BY paciente_cpf, cod_exame ORDER BY data_exame DESC ) = 1 ),

exames_pivot AS ( SELECT paciente_cpf AS cpf, MAX(CASE WHEN cod_exame = ‘CRE’ THEN resultado_num END) AS creatinina, MAX(CASE WHEN cod_exame = ‘COL’ THEN resultado_num END) AS colesterol_total, MAX(CASE WHEN cod_exame = ‘HDL’ THEN resultado_num END) AS hdl, MAX(CASE WHEN cod_exame = ‘LDL’ THEN resultado_num END) AS ldl, MAX(CASE WHEN cod_exame = ‘TRI’ THEN resultado_num END) AS triglicerides, COALESCE( MAX(CASE WHEN cod_exame = ‘GLI’ THEN resultado_num END), MAX(CASE WHEN cod_exame = ‘GLU’ THEN resultado_num END) ) AS glicemia, MAX(CASE WHEN cod_exame = ‘HGL’ THEN resultado_num END) AS hba1c,

    -- DATAS dos exames
    MAX(CASE WHEN cod_exame = 'CRE' THEN data_exame END) AS data_creatinina,
    MAX(CASE WHEN cod_exame = 'COL' THEN data_exame END) AS data_colesterol,
    MAX(CASE WHEN cod_exame = 'HDL' THEN data_exame END) AS data_hdl,
    MAX(CASE WHEN cod_exame = 'LDL' THEN data_exame END) AS data_ldl,
    COALESCE(  -- ← ADICIONAR ESTA LINHA
        MAX(CASE WHEN cod_exame = 'GLI' THEN data_exame END),
        MAX(CASE WHEN cod_exame = 'GLU' THEN data_exame END)
    ) AS data_glicemia,
    MAX(CASE WHEN cod_exame = 'HGL' THEN data_exame END) AS data_hba1c,  -- ← ADICIONAR ESTA LINHA
    
    -- DIAS desde último exame geral
    DATE_DIFF(
        CURRENT_DATE(),
        GREATEST(
            IFNULL(MAX(CASE WHEN cod_exame = 'CRE' THEN data_exame END), DATE '1900-01-01'),
            IFNULL(MAX(CASE WHEN cod_exame = 'COL' THEN data_exame END), DATE '1900-01-01'),
            IFNULL(MAX(CASE WHEN cod_exame = 'HDL' THEN data_exame END), DATE '1900-01-01'),
            IFNULL(MAX(CASE WHEN cod_exame = 'LDL' THEN data_exame END), DATE '1900-01-01')
        ),
        DAY
    ) AS dias_desde_ultimo_exame,
    
    -- ← ADICIONAR: Dias desde última glicemia
    DATE_DIFF(CURRENT_DATE(), COALESCE(
        MAX(CASE WHEN cod_exame = 'GLI' THEN data_exame END),
        MAX(CASE WHEN cod_exame = 'GLU' THEN data_exame END)
    ), DAY) AS dias_desde_ultima_glicemia
    
FROM exames_mais_recentes
GROUP BY paciente_cpf

),

– ============================================================================ – SEÇÃO 3: DADOS DEMOGRÁFICOS E DE CADASTRO (FONTE: FICHA_A + PACIENTE) – ============================================================================

ultima_ficha AS ( SELECT * FROM rj-sms.brutos_prontuario_vitacare.ficha_a WHERE cadastro_permanente = TRUE AND (situacao_usuario = ‘Ativo’ OR situacao_usuario IS NULL) AND cpf IS NOT NULL AND obito = FALSE AND NOT REGEXP_CONTAINS(UPPER(COALESCE(equipe_nome, ’‘)), r’ESB’) AND ine_equipe IS NOT NULL QUALIFY ROW_NUMBER() OVER ( PARTITION BY cpf ORDER BY data_atualizacao_cadastro DESC ) = 1 ),

dados_paciente AS ( SELECT f.cpf, f.nome AS nome, f.sexo AS genero, f.data_nascimento, DATE_DIFF(CURRENT_DATE(), f.data_nascimento, YEAR) AS idade, f.raca_cor AS raca,

    -- ========================================
    -- DADOS DE CADASTRO ESF
    -- ========================================
    f.ine_equipe AS id_ine_cadastro,
    f.equipe_nome AS nome_esf_cadastro,
    f.unidade_cadastro AS id_cnes_cadastro,
    f.unidade_nome AS nome_clinica_cadastro,
    f.ap_cadastro AS area_programatica_cadastro,
    
    -- Profissionais da ESF (da tabela paciente)
    p.equipe_saude_familia[SAFE_OFFSET(0)].medicos[SAFE_OFFSET(0)].nome 
        AS nome_medico_esf_cadastro,
    p.equipe_saude_familia[SAFE_OFFSET(0)].enfermeiros[SAFE_OFFSET(0)].nome 
        AS nome_enfermeiro_esf_cadastro,
    
    -- ========================================
    -- DADOS DE CADASTRO - DATAS
    -- ========================================
    f.data_cadastro,
    f.data_atualizacao_cadastro,
    f.situacao_usuario,
    
    -- ========================================
    -- DETERMINANTES SOCIAIS
    -- ========================================
    f.escolaridade,
    f.renda_familiar,
    f.situacao_profissional,
    f.ocupacao,
    f.vulnerabilidade_social,
    f.em_situacao_de_rua,
    f.possui_plano_saude,
    f.familia_beneficiaria_cfc,
    f.familia_beneficiaria_auxilio_brasil,
    
    -- ========================================
    -- IDENTIDADE E DADOS PESSOAIS
    -- ========================================
    f.nome_social,
    f.identidade_genero,
    f.orientacao_sexual,
    f.nome_mae,
    f.nome_pai,
    f.nacionalidade,
    f.pais_nascimento,
    f.estado_nascimento,
    f.religiao,
    f.telefone,
    
    -- ========================================
    -- ENDEREÇO E TERRITÓRIO
    -- ========================================
    f.bairro,
    f.logradouro,
    f.tipo_logradouro,
    f.microarea,
    f.municipio_residencia.nome AS municipio_residencia,
    f.estado_residencia,
    
    -- ========================================
    -- CONDIÇÕES DE MORADIA E SANEAMENTO
    -- ========================================
    f.tipo_domicilio,
    f.situacao_moradia_posse,
    f.comodos,
    f.abastecimento_agua,
    f.tratamento_agua,
    f.possui_filtro_agua,
    f.esgotamento_sanitario,
    f.destino_lixo,
    f.luz_eletrica,
    f.animais_no_domicilio,
    
    -- ========================================
    -- SITUAÇÃO FAMILIAR E COMUNITÁRIA
    -- ========================================
    f.situacao_familiar,
    f.responsavel_familiar,
    f.relacao_responsavel_familiar,
    f.participa_grupo_comunitario,
    f.membro_comunidade_tradicional,
    f.territorio_social,
    
    -- ========================================
    -- EDUCAÇÃO E SAÚDE (CRIANÇAS/JOVENS)
    -- ========================================
    f.frequenta_escola,
    f.crianca_matriculada_creche_pre_escola,
    
    -- ========================================
    -- CONDIÇÕES DE SAÚDE AUTORREFERIDAS (FICHA A)
    -- ========================================
    ARRAY_TO_STRING(f.doencas_condicoes, '; ') AS doencas_condicoes_ficha_a
    
FROM ultima_ficha f
LEFT JOIN `rj-sms.saude_historico_clinico.paciente` p
    ON f.cpf = p.cpf

),

– ============================================================================ – SEÇÃO 4: ÚLTIMAS MEDIDAS ANTROPOMÉTRICAS - INTEGRAÇÃO VITACARE + EPISODIO – ============================================================================

ultimas_medidas_antropometricas AS ( SELECT cpf, peso, altura, IMC, obesidade_por_IMC, data_peso, data_altura, fonte FROM ( SELECT cpf, peso, altura, IMC, obesidade_por_IMC, data_peso, data_altura, fonte, ROW_NUMBER() OVER ( PARTITION BY cpf ORDER BY COALESCE(data_peso, data_altura) DESC – ← Prioriza mais recente ) AS rn_final FROM ( – Vitacare SELECT cpf, peso_fix AS peso, altura_fix AS altura, imc_fix AS IMC, CASE WHEN imc_fix >= 30 THEN TRUE ELSE FALSE END AS obesidade_por_IMC, CAST(datahora_inicio AS DATE) AS data_peso, CAST(datahora_inicio AS DATE) AS data_altura, ‘vitacare’ AS fonte FROM rj-sms-sandbox.sub_pav_us.indicadores_vitacare_fix WHERE peso_fix IS NOT NULL OR altura_fix IS NOT NULL

        UNION ALL
        
        -- Episodio
        SELECT
            paciente_cpf AS cpf,
            peso,
            altura,
            IMC,
            obesidade_por_IMC,
            data_peso,
            data_altura,
            'episodio_assistencial' AS fonte
        FROM (
            SELECT
                paciente_cpf,
                
                FIRST_VALUE(medidas.peso IGNORE NULLS) OVER (
                    PARTITION BY paciente_cpf 
                    ORDER BY 
                        CASE WHEN medidas.peso IS NOT NULL THEN entrada_data END DESC NULLS LAST
                ) AS peso,
                
                FIRST_VALUE(
                    CASE WHEN medidas.peso IS NOT NULL THEN entrada_data END IGNORE NULLS
                ) OVER (
                    PARTITION BY paciente_cpf 
                    ORDER BY 
                        CASE WHEN medidas.peso IS NOT NULL THEN entrada_data END DESC NULLS LAST
                ) AS data_peso,
                
                FIRST_VALUE(medidas.altura IGNORE NULLS) OVER (
                    PARTITION BY paciente_cpf 
                    ORDER BY 
                        CASE WHEN medidas.altura IS NOT NULL THEN entrada_data END DESC NULLS LAST
                ) AS altura,
                
                FIRST_VALUE(
                    CASE WHEN medidas.altura IS NOT NULL THEN entrada_data END IGNORE NULLS
                ) OVER (
                    PARTITION BY paciente_cpf 
                    ORDER BY 
                        CASE WHEN medidas.altura IS NOT NULL THEN entrada_data END DESC NULLS LAST
                ) AS data_altura,
                
                CASE 
                    WHEN medidas.peso IS NOT NULL AND medidas.altura IS NOT NULL AND medidas.altura > 0
                    THEN ROUND(medidas.peso / POWER(medidas.altura, 2), 1)
                    ELSE NULL
                END AS IMC,
                
                CASE 
                    WHEN medidas.peso IS NOT NULL AND medidas.altura IS NOT NULL AND medidas.altura > 0
                         AND (medidas.peso / POWER(medidas.altura, 2)) >= 30
                    THEN TRUE
                    ELSE FALSE
                END AS obesidade_por_IMC,
                
                ROW_NUMBER() OVER (PARTITION BY paciente_cpf ORDER BY entrada_data DESC) AS rn
                
            FROM `rj-sms.saude_historico_clinico.episodio_assistencial`
            WHERE medidas.peso IS NOT NULL OR medidas.altura IS NOT NULL
        )
        WHERE rn = 1
    )
)
WHERE rn_final = 1  -- ← AGORA pega apenas 1 total por CPF

),

– ========================= – SEÇÃO 6: MORBIDADES POR CID – ========================= morbidades_base AS ( SELECT REGEXP_REPLACE(CAST(ea.paciente_cpf AS STRING), r’‘,’’) AS cpf, ea.entrada_data, cond.id AS cid FROM rj-sms.saude_historico_clinico.episodio_assistencial ea, UNNEST(ea.condicoes) AS cond WHERE ea.paciente_cpf IS NOT NULL AND cond.id IS NOT NULL ),

– ============================================================================ – SEÇÃO 6: MORBIDADES POR CID (CONFORME ESPECIFICAÇÃO) – ============================================================================

morbidades_por_cid AS ( SELECT cpf,

    -- ========================================
    -- SEÇÃO 1: MORBIDADES PRINCIPAIS
    -- ========================================
    MIN(CASE WHEN REGEXP_CONTAINS(cid, r'^(I1[0-6]|O1[01])') THEN entrada_data END) AS HAS,
    MIN(CASE WHEN REGEXP_CONTAINS(cid, r'^(E1[0-4]|O24|H360|H280|G632|G590|G990|N083|M142)') THEN entrada_data END) AS DM,
    MIN(CASE WHEN REGEXP_CONTAINS(cid, r'^E10') THEN cid END) AS DM_cid_raw,
    
    -- ========================================
    -- SEÇÃO 2: CARDIOVASCULARES
    -- ========================================
    MIN(CASE WHEN REGEXP_CONTAINS(cid, r'^(I2[0-5]|Z95[15])') THEN entrada_data END) AS CI,
    MIN(CASE WHEN REGEXP_CONTAINS(cid, r'^(I110|I13[02]|I42[0-9]|I43|I50|I517|P290)') THEN entrada_data END) AS ICC,
    MIN(CASE WHEN REGEXP_CONTAINS(cid, r'^(I6|G4[56]|H340)') THEN entrada_data END) AS stroke,
    MIN(CASE WHEN REGEXP_CONTAINS(cid, r'^(I44[1-3]|I45[69]|I4[7-9]|R00[018]|T821|Z45[05]|Z950)') THEN entrada_data END) AS arritmia,
    MIN(CASE WHEN REGEXP_CONTAINS(cid, r'^(A520|I0[5-9]|I34|I3[5-9]|Q23[0-3]|Z95[2-4])') THEN entrada_data END) AS valvular,
    MIN(CASE WHEN REGEXP_CONTAINS(cid, r'^(I2[6-8])') THEN entrada_data END) AS circ_pulm,
    MIN(CASE WHEN REGEXP_CONTAINS(cid, r'^(I7[01]|I73[189]|I77[12]|I79[02]|K55[189]|Z95[89])') THEN entrada_data END) AS vascular_periferica,
    
    -- ========================================
    -- SEÇÃO 3: NEUROLÓGICAS
    -- ========================================
    MIN(CASE WHEN REGEXP_CONTAINS(cid, r'^(G4[01])') THEN entrada_data END) AS epilepsy,
    MIN(CASE WHEN REGEXP_CONTAINS(cid, r'^(G2[0-2])') THEN entrada_data END) AS parkinsonism,
    MIN(CASE WHEN REGEXP_CONTAINS(cid, r'^G35') THEN entrada_data END) AS multiple_sclerosis,
    MIN(CASE WHEN REGEXP_CONTAINS(cid, r'^(G1[0-3]|G25[45]|G31[012]|G3[2367]|G93[14]|R470|R56)') THEN entrada_data END) AS neuro,
    MIN(CASE WHEN REGEXP_CONTAINS(cid, r'^(F0[0-35]|G30)') THEN entrada_data END) AS dementia,
    MIN(CASE WHEN REGEXP_CONTAINS(cid, r'^(G041|G114|G80[12]|G8[1-3])') THEN entrada_data END) AS plegia,
    
    -- ========================================
    -- SEÇÃO 4: PSIQUIÁTRICAS
    -- ========================================
    MIN(CASE WHEN REGEXP_CONTAINS(cid, r'^(F2[0-9]|F31)') THEN entrada_data END) AS psicoses,
    MIN(CASE WHEN REGEXP_CONTAINS(cid, r'^(F3[2-49]|F4[01])') THEN entrada_data END) AS depre_ansiedade,
    
    -- ========================================
    -- SEÇÃO 5: PULMONARES
    -- ========================================
    MIN(CASE WHEN REGEXP_CONTAINS(cid, r'^(I27[89]|J4[0-4]|J47|J6[0-7]|J684|J70[13])') THEN entrada_data END) AS COPD,
    MIN(CASE WHEN REGEXP_CONTAINS(cid, r'^(J4[56])') THEN entrada_data END) AS asthma,
    
    -- ========================================
    -- SEÇÃO 6: NEOPLASIAS
    -- ========================================
    MIN(CASE WHEN REGEXP_CONTAINS(cid, r'^C50') THEN entrada_data END) AS neoplasia_mama,
    MIN(CASE WHEN REGEXP_CONTAINS(cid, r'^C53') THEN entrada_data END) AS neoplasia_colo_uterino,
    MIN(CASE WHEN REGEXP_CONTAINS(cid, r'^(C5[1-28])') THEN entrada_data END) AS neoplasia_feminina_estrita,
    MIN(CASE WHEN REGEXP_CONTAINS(cid, r'^(C6[0-3])') THEN entrada_data END) AS neoplasia_masculina_estrita,
    MIN(CASE WHEN REGEXP_CONTAINS(cid, r'^(C9[1-5])') THEN entrada_data END) AS leukemia,
    MIN(CASE WHEN REGEXP_CONTAINS(cid, r'^(C8[1-6]|C96)') THEN entrada_data END) AS lymphoma,
    MIN(CASE WHEN REGEXP_CONTAINS(cid, r'^(C7[7-9]|C80)') THEN entrada_data END) AS metastasis,
    MIN(CASE WHEN REGEXP_CONTAINS(cid, r'^(C[0-4][0-9]|C[567][0-9])') 
             AND NOT REGEXP_CONTAINS(cid, r'^(C44|C50|C5[1-8]|C6[0-3]|C7[7-9]|C80|C8[1-6]|C9[1-6])') 
        THEN entrada_data END) AS neoplasia_ambos_os_sexos,
    
    -- ========================================
    -- SEÇÃO 7: RENAIS E METABÓLICAS
    -- ========================================
    MIN(CASE WHEN REGEXP_CONTAINS(cid, r'^(I120|I131|N03[2-7]|N05[2-7]|N1[89]|N250|Z49[0-2]|Z940|Z992)') THEN entrada_data END) AS IRC_por_CID,
    MIN(CASE WHEN REGEXP_CONTAINS(cid, r'^E78') THEN entrada_data END) AS dislipidemia_por_CID,
    MIN(CASE WHEN REGEXP_CONTAINS(cid, r'^E66') THEN entrada_data END) AS obesidade,
    MIN(CASE WHEN REGEXP_CONTAINS(cid, r'^(E0[0-3]|E05|E06|E890)') THEN entrada_data END) AS tireoide,
    
    -- ========================================
    -- SEÇÃO 8: HEMATOLÓGICAS
    -- ========================================
    MIN(CASE WHEN REGEXP_CONTAINS(cid, r'^(D6[5-9])') THEN entrada_data END) AS coagulo,

    -- HAS - Consolidado + Detalhamento
    -- ========================================
    m.HAS AS has_por_cid,
    cm_has.has_por_medida_critica,

            m.HAS,
            cm_has.has_por_medida_critica,
            cm_has.has_por_medidas_repetidas,
            cmed_has.has_por_medicamento
        ]) AS data
        WHERE data IS NOT NULL

– Última medição de HbA1c com valor anterior hba1c_progressao AS ( SELECT cpf, hba1c AS hba1c_atual, data_coleta AS data_hba1c_atual, LAG(hba1c) OVER (PARTITION BY cpf ORDER BY data_coleta, id_hci) AS hba1c_anterior, LAG(data_coleta) OVER (PARTITION BY cpf ORDER BY data_coleta) AS data_hba1c_anterior FROM todas_medidas_glicemia WHERE hba1c IS NOT NULL QUALIFY ROW_NUMBER() OVER (PARTITION BY cpf ORDER BY data_coleta DESC, id_hci DESC) = 1 ),

– Calcular tendência e status progressao_glicemica AS ( SELECT hp.cpf, hp.hba1c_atual, hp.data_hba1c_atual, hp.hba1c_anterior, hp.data_hba1c_anterior,

    -- Intervalo entre medições
    DATE_DIFF(hp.data_hba1c_atual, hp.data_hba1c_anterior, DAY) AS dias_entre_hba1c,
    
    -- Meta de HbA1c por idade
    CASE
        WHEN dp.idade < 60 THEN 7.0
        ELSE 7.5
    END AS meta_hba1c,
    
    -- Status de controle
    CASE
        WHEN dc.DM IS NULL THEN  -- Paciente NÃO tem diabetes
            CASE
                WHEN hp.hba1c_atual IS NULL THEN 'sem_dados'
                WHEN hp.hba1c_atual < 5.7 THEN 'rastreio - normal'
                WHEN hp.hba1c_atual < 6.5 THEN 'rastreio - pre_diabetes'
                ELSE 'rastreio - provável_diabetes'
            END
        WHEN dc.DM IS NOT NULL THEN  -- Paciente TEM diabetes
            CASE
                WHEN hp.hba1c_atual IS NULL THEN 'sem_dados'
                WHEN dp.idade >= 70 AND hp.hba1c_atual < 6.0 THEN 'abaixo_da_meta'
                WHEN hp.hba1c_atual <= CASE WHEN dp.idade < 60 THEN 7.0 ELSE 7.5 END THEN 'dentro_da_meta'
                ELSE 'fora_da_meta'
            END
        ELSE 'sem_dados'
    END AS status_controle_glicemico,
    
    -- Tendência da HbA1c - LÓGICA AJUSTADA
    CASE
        -- PACIENTES SEM DIABETES: Apenas rastreio
        WHEN dc.DM IS NULL THEN
            CASE
                WHEN hp.hba1c_atual IS NULL THEN 'sem_dados'
                WHEN hp.hba1c_atual < 5.7 THEN 'rastreio - normal'
                WHEN hp.hba1c_atual < 6.5 THEN 'rastreio - pre_diabetes'
                ELSE 'rastreio - provável_diabetes'
            END
        
        -- PACIENTES COM DIABETES: Tendência baseada no status
        WHEN dc.DM IS NOT NULL THEN
            CASE
                WHEN hp.hba1c_atual IS NULL THEN 'sem_dados'
                
                -- Se está DENTRO DA META → sempre estável ou sem referência
                WHEN hp.hba1c_atual <= CASE WHEN dp.idade < 60 THEN 7.0 ELSE 7.5 END THEN
                    CASE
                        WHEN hp.hba1c_anterior IS NULL THEN 'estavel'
                        ELSE 'estavel'
                    END
                
                -- Se está ABAIXO DA META (idosos) → sempre estável ou sem referência
                WHEN dp.idade >= 70 AND hp.hba1c_atual < 6.0 THEN
                    CASE
                        WHEN hp.hba1c_anterior IS NULL THEN 'estavel'
                        ELSE 'estavel'
                    END
                
                -- Se está FORA DA META → calcular tendência (melhorando/piorando/estável)
                ELSE
                    CASE
                        WHEN hp.hba1c_anterior IS NULL THEN 'sem_referencia_previa'
                        WHEN hp.hba1c_atual < (hp.hba1c_anterior - 0.4) THEN 'melhorando'
                        WHEN hp.hba1c_atual > (hp.hba1c_anterior + 0.4) THEN 'piorando'
                        ELSE 'estavel'
                    END
            END
        
        ELSE 'sem_dados'
    END AS tendencia_hba1c
    
FROM hba1c_progressao hp
LEFT JOIN dados_paciente dp ON hp.cpf = dp.cpf
LEFT JOIN has_dm_consolidado dc ON hp.cpf = dc.cpf

– ============================================================================ – SEÇÃO: HISTÓRICO DE CONTROLE GLICÊMICO – ============================================================================

historico_controle_glicemia AS ( SELECT glic.cpf, glic.data_coleta, glic.hba1c,

    -- Data da próxima medição (ou CURRENT_DATE se for a última)
    COALESCE(
        LEAD(glic.data_coleta) OVER (PARTITION BY glic.cpf ORDER BY glic.data_coleta),
        CURRENT_DATE()
    ) AS proxima_data,
    
    -- Dias até próxima medição
    DATE_DIFF(
        COALESCE(
            LEAD(glic.data_coleta) OVER (PARTITION BY glic.cpf ORDER BY glic.data_coleta),
            CURRENT_DATE()
        ),
        glic.data_coleta,
        DAY
    ) AS dias_ate_proxima,
    
    -- Meta de HbA1c por idade
    CASE
        WHEN dp.idade < 60 THEN 7.0
        ELSE 7.5
    END AS meta_hba1c,
    
    -- Status de controle nesse período
    CASE
        WHEN glic.hba1c <= CASE WHEN dp.idade < 60 THEN 7.0 ELSE 7.5 END
        THEN 'controlado'
        ELSE 'descontrolado'
    END AS status_periodo
    
FROM todas_medidas_glicemia glic
INNER JOIN dados_paciente dp ON glic.cpf = dp.cpf
INNER JOIN has_dm_consolidado hc ON glic.cpf = hc.cpf
WHERE hc.DM IS NOT NULL  -- Só pacientes COM DM
  AND glic.hba1c IS NOT NULL  -- Só medidas com HbA1c
  AND glic.data_coleta >= hc.DM  -- Só medidas APÓS diagnóstico
  
UNION ALL

-- ADICIONAR: Período do diagnóstico até a primeira medida
SELECT
    hc.cpf,
    hc.DM AS data_coleta,  -- Data do diagnóstico
    primeira_hba1c.hba1c,
    primeira_hba1c.data_coleta AS proxima_data,  -- Data da primeira medida
    DATE_DIFF(primeira_hba1c.data_coleta, hc.DM, DAY) AS dias_ate_proxima,
    CASE WHEN dp.idade < 60 THEN 7.0 ELSE 7.5 END AS meta_hba1c,
    CASE
        WHEN primeira_hba1c.hba1c <= CASE WHEN dp.idade < 60 THEN 7.0 ELSE 7.5 END
        THEN 'controlado'
        ELSE 'descontrolado'
    END AS status_periodo
FROM has_dm_consolidado hc
INNER JOIN dados_paciente dp ON hc.cpf = dp.cpf
INNER JOIN (
    -- Primeira medida de HbA1c após diagnóstico
    SELECT
        glic.cpf,
        glic.data_coleta,
        glic.hba1c,
        ROW_NUMBER() OVER (PARTITION BY glic.cpf ORDER BY glic.data_coleta) AS rn
    FROM todas_medidas_glicemia glic

    WHERE hc2.DM IS NOT NULL
      AND glic.hba1c IS NOT NULL
      AND glic.data_coleta >= hc2.DM
) primeira_hba1c ON hc.cpf = primeira_hba1c.cpf AND primeira_hba1c.rn = 1

  AND DATE_DIFF(primeira_hba1c.data_coleta, hc.DM, DAY) > 0  -- Só se houver GAP

– Agregação por paciente – Agregação por paciente - APENAS DIAS resumo_controle_glicemia_dias AS ( SELECT cpf,

    -- ========================================
    -- DESDE O DIAGNÓSTICO
    -- ========================================
    SUM(CASE WHEN status_periodo = 'controlado' THEN dias_ate_proxima ELSE 0 END) AS dias_dm_controlado,
    SUM(CASE WHEN status_periodo = 'descontrolado' THEN dias_ate_proxima ELSE 0 END) AS dias_dm_descontrolado,
    
    -- Dias desde última HbA1c
    DATE_DIFF(CURRENT_DATE(), MAX(data_coleta), DAY) AS dias_desde_ultima_hba1c,
    
    -- ========================================
    -- ÚLTIMOS 365 DIAS
    -- ========================================
    SUM(CASE 
        WHEN status_periodo = 'controlado' THEN
            CASE
                WHEN data_coleta >= DATE_SUB(CURRENT_DATE(), INTERVAL 365 DAY)
                     AND proxima_data <= CURRENT_DATE()
                THEN dias_ate_proxima
                
                WHEN data_coleta < DATE_SUB(CURRENT_DATE(), INTERVAL 365 DAY)
                     AND proxima_data >= DATE_SUB(CURRENT_DATE(), INTERVAL 365 DAY)
                THEN DATE_DIFF(
                    LEAST(proxima_data, CURRENT_DATE()),
                    DATE_SUB(CURRENT_DATE(), INTERVAL 365 DAY),
                    DAY
                )

intervalos_consultas_raw AS ( SELECT ea.paciente_cpf AS cpf, ea.entrada_data,

    DATE_DIFF(
        ea.entrada_data,
        LAG(ea.entrada_data) OVER (PARTITION BY ea.paciente_cpf ORDER BY ea.entrada_data),
        DAY
    ) AS intervalo_dias

INNER JOIN dados_paciente dp ON ea.paciente_cpf = dp.cpf
WHERE ea.paciente_cpf IS NOT NULL
  AND ea.profissional_saude_responsavel.especialidade IS NOT NULL
  AND DATE_DIFF(CURRENT_DATE(), ea.entrada_data, DAY) <= 365

),

intervalos_consultas AS ( SELECT cpf, APPROX_QUANTILES(intervalo_dias, 2)[SAFE_OFFSET(1)] AS intervalo_mediano_dias, ROUND(AVG(intervalo_dias), 1) AS intervalo_medio_dias, ARRAY_AGG(intervalo_dias ORDER BY entrada_data DESC LIMIT 1)[SAFE_OFFSET(0)] AS ultimo_intervalo_dias, COUNT(intervalo_dias) AS n_intervalos FROM intervalos_consultas_raw WHERE intervalo_dias IS NOT NULL GROUP BY cpf ),

        CASE WHEN hdl IS NOT NULL THEN CONCAT('; HDL: ', CAST(ROUND(hdl) AS STRING)) ELSE '' END,
        CASE WHEN pressao_sistolica IS NOT NULL THEN CONCAT('; PAS: ', CAST(ROUND(pressao_sistolica) AS STRING)) ELSE '' END,
        CASE WHEN DM IS NOT NULL THEN '; DM: sim' ELSE '' END,
        CASE WHEN tabaco IS NOT NULL THEN '; Tabagismo: sim' ELSE '' END
    ) AS variaveis_usadas_calculo,
    
    -- ========================================
    -- VARIÁVEIS AUSENTES NO CÁLCULO
    -- ========================================
    CONCAT(
        CASE WHEN colesterol_total IS NULL THEN 'Colesterol total; ' ELSE '' END,
        CASE WHEN hdl IS NULL THEN 'HDL; ' ELSE '' END,
        CASE WHEN pressao_sistolica IS NULL THEN 'Pressão arterial; ' ELSE '' END,
        CASE WHEN tabaco IS NULL THEN 'História de tabagismo; ' ELSE '' END
    ) AS variaveis_ausentes_calculo,
    
    -- ========================================
    -- RECLASSIFICAÇÃO POR CONDIÇÕES CLÍNICAS
    -- ========================================
    
    -- Determina se há reclassificação
    CASE
        -- MUITO ALTO: Aterosclerose significativa
        WHEN CI IS NOT NULL OR stroke IS NOT NULL OR vascular_periferica IS NOT NULL THEN 'reclassificado_muito_alto'
        
        -- ALTO: DRC, eGFR < 60, LDL muito elevado
        WHEN IRC IS NOT NULL OR egfr < 60 OR ldl >= 190 THEN 'reclassificado_alto'
        
        -- DM sem aterosclerose: pode ser intermediário ou usar Framingham
        WHEN DM IS NOT NULL THEN 'diabetes_avaliar_framingham'
        
        -- Sem reclassificação: usar Framingham puro
        ELSE 'usar_framingham'
    END AS tipo_estratificacao,
    
    -- ========================================
    -- PERCENTUAL FINAL (STRING FORMATADA)
    -- ========================================
    CASE
        -- MUITO ALTO: >30%
        WHEN CI IS NOT NULL OR stroke IS NOT NULL OR vascular_periferica IS NOT NULL THEN '>30%'
        
        -- ALTO: >20% (homens) ou >10% (mulheres)
        WHEN IRC IS NOT NULL OR egfr < 60 OR ldl >= 190 THEN
            CASE WHEN genero = 'masculino' THEN '>20%' ELSE '>10%' END
        
        -- DIABETES: avaliar Framingham
        WHEN DM IS NOT NULL THEN
            CASE
                -- Se Framingham já indica alto risco
                WHEN genero = 'masculino' AND percentual_risco_10_anos > 20 THEN '>20%'
                WHEN genero = 'feminino' AND percentual_risco_10_anos > 10 THEN '>10%'
                -- Senão, usar valor calculado mas classificar como intermediário
                ELSE CONCAT(CAST(ROUND(percentual_risco_10_anos, 1) AS STRING), '%')
            END
        
        -- SEM CONDIÇÕES: Usar Framingham calculado
        WHEN percentual_risco_10_anos IS NOT NULL THEN

        ELSE 'Não calculável'

    -- ========================================
    -- CATEGORIA DE RISCO FINAL
    -- ========================================
    CASE
        -- MUITO ALTO
        WHEN CI IS NOT NULL OR stroke IS NOT NULL OR vascular_periferica IS NOT NULL THEN 'MUITO ALTO'
        
        -- ALTO
        WHEN IRC IS NOT NULL OR egfr < 60 OR ldl >= 190 THEN 'ALTO'
        WHEN genero = 'masculino' AND percentual_risco_10_anos > 20 THEN 'ALTO'
        WHEN genero = 'feminino' AND percentual_risco_10_anos > 10 THEN 'ALTO'
        
        -- INTERMEDIÁRIO

        WHEN genero = 'masculino' AND percentual_risco_10_anos BETWEEN 5 AND 20 THEN 'INTERMEDIÁRIO'
        WHEN genero = 'feminino' AND percentual_risco_10_anos BETWEEN 5 AND 10 THEN 'INTERMEDIÁRIO'
        
        -- BAIXO
        WHEN percentual_risco_10_anos < 5 THEN 'BAIXO'
        
        ELSE 'NÃO CLASSIFICADO'

    -- MOTIVO DA RECLASSIFICAÇÃO
    -- ========================================

        WHEN CI IS NOT NULL THEN 'Cardiopatia isquêmica prévia'
        WHEN stroke IS NOT NULL THEN 'AVC/Stroke prévio'
        WHEN vascular_periferica IS NOT NULL THEN 'Doença arterial periférica'
```
